            <button class="main-btn btn-apply" style="background:var(--gold); color:#000" onclick="processData('big')">TÔ SỐ LỚN</button>
            <button class="main-btn btn-apply" style="background:var(--green); color:#000" onclick="processData('small')">TÔ SỐ NHỎ</button>

<script>
    const fileInput = document.getElementById('fileInput');
    const rowLengthInput = document.getElementById('rowLength');
    const t1Input = document.getElementById('t1Input');
    const headerGrid = document.getElementById('header-grid');
    const resultGrid = document.getElementById('result-grid');
    const resultContainer = document.getElementById('result-container');
    const headerWrapper = document.getElementById('headerWrapper');

    let currentNumbers = []; 

    resultContainer.addEventListener('scroll', () => {
        headerWrapper.scrollLeft = resultContainer.scrollLeft;
    });

    window.onload = function() {
        rowLengthInput.value = localStorage.getItem('userRowLength') || 10;
        t1Input.value = localStorage.getItem('userT1') || 'AAAb';
        const savedData = localStorage.getItem('userCurrentNumbers');
        if (savedData) {
            currentNumbers = JSON.parse(savedData);
            refreshAll();
        }
        updateHeaderGrid();
    };

    function clearMemory() {
        localStorage.clear(); location.reload();
    }

    function refreshAll() {
        const x = parseInt(rowLengthInput.value) || 10;
        const mode = localStorage.getItem('userLastMode') || 't1';
        updateHeaderGrid();
        renderGrid(currentNumbers, x, mode);
    }

    function applyT1() {
        localStorage.setItem('userT1', t1Input.value.trim());
        localStorage.setItem('userLastMode', 't1');
        refreshAll();
    }

    function processData(mode) {
        localStorage.setItem('userLastMode', mode);
        refreshAll();
    }

    fileInput.addEventListener('change', function() {
        const reader = new FileReader();
        reader.onload = function(e) {
            const cleanText = e.target.result.replace(/[^0-9]/g, '');
            currentNumbers = cleanText.split('').map(Number);
            localStorage.setItem('userCurrentNumbers', JSON.stringify(currentNumbers));
            refreshAll();
        };
        reader.readAsText(this.files[0]);
    });

    rowLengthInput.addEventListener('input', () => {
        localStorage.setItem('userRowLength', rowLengthInput.value);
        refreshAll();
    });

    function updateHeaderGrid() {
        const x = parseInt(rowLengthInput.value) || 10;
        headerGrid.style.gridTemplateColumns = `var(--stt-width) var(--spacer-width) repeat(${x}, var(--cell-size))`;
        headerGrid.innerHTML = '';
        
        const sttEmpty = document.createElement('div');
        sttEmpty.className = 'row-stt';
        sttEmpty.style.background = 'transparent';
        headerGrid.appendChild(sttEmpty);

        const spacerEmpty = document.createElement('div');
        spacerEmpty.className = 'spacer-cell';
        headerGrid.appendChild(spacerEmpty);

        for (let i = 0; i < x; i++) {
            const box = document.createElement('div');
            box.className = `index-box idx-${i % 10}`;
            box.textContent = i % 10;
            headerGrid.appendChild(box);
        }
    }

    function renderGrid(data, columns, mode) {
        resultGrid.innerHTML = ''; 
        if (data.length === 0) return;
        
        resultGrid.style.gridTemplateColumns = `var(--stt-width) var(--spacer-width) repeat(${columns}, var(--cell-size))`;
        
        const T1 = t1Input.value.trim().toUpperCase() || 'AAAb';
        const t1Len = T1.length;
        const isYellow = new Array(data.length).fill(false);

        if (mode === 't1') {
            const reversedMap = data.map((num, i) => {
                const char = T1[i % t1Len];
                return char === 'A' ? num < 5 : num >= 5;
            });
            for (let i = 0; i <= data.length - 5; i++) {
                if (reversedMap.slice(i, i+5).every(v => v)) {
                    for (let j = 0; j < 5; j++) isYellow[i+j] = true;
                }
            }
        }

        const fragment = document.createDocumentFragment();
        data.forEach((num, index) => {
            if (index % columns === 0) {
                const sttBox = document.createElement('div');
                sttBox.className = 'row-stt';
                sttBox.textContent = (index / columns) + 1;
                fragment.appendChild(sttBox);
                
                const spacer = document.createElement('div');
                spacer.className = 'spacer-cell';
                fragment.appendChild(spacer);
            }

            const box = document.createElement('div');
            box.className = 'number-box';
            box.textContent = num;
            
            if (mode === 't1' && isYellow[index]) box.classList.add('high');
            else if (mode === 'big' && num >= 5) box.classList.add('high');
            else if (mode === 'small' && num < 5) box.classList.add('low');
            
            fragment.appendChild(box);
        });
        resultGrid.appendChild(fragment);
    }
</script>
