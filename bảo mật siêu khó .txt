<script>
    let selShapes = [];
    let selSpecs = [];

    const specs = ['!', '@', '#', '$', '%', '^', '&', '*', '(', ')', '-', '+', '=', '{', '}', '[', ']', ';', ':', '?', '/', '>', '<', '~', '|'];
    const icons = [
        '<circle cx="12" cy="12" r="9"/>','<rect x="4" y="4" width="16" height="16"/>',
        '<path d="M12 2 L22 20 L2 20 Z"/>','<path d="M12 2 L15 9 L22 9 L17 14 L19 21 L12 17 L5 21 L7 14 L2 9 L9 9 Z"/>',
        '<ellipse cx="12" cy="12" rx="9" ry="5"/>','<path d="M3 12 h18 M12 3 v18"/>',
        '<path d="M5 5 L19 19 M19 5 L5 19"/>','<circle cx="12" cy="12" r="4"/>',
        '<path d="M12 21 a9 9 0 1 0 0 -18 v18"/>','<rect x="8" y="2" width="8" height="20" rx="2"/>'
    ];

    const sGrid = document.getElementById('specGrid');
    specs.forEach(s => {
        const d = document.createElement('div');
        d.className = 'item-select';
        d.innerText = s;
        d.onclick = () => {
            if(selSpecs.includes(s)) selSpecs = selSpecs.filter(x => x !== s);
            else selSpecs.push(s);
            d.classList.toggle('active');
        };
        sGrid.appendChild(d);
    });

    const gGrid = document.getElementById('shapeGrid');
    icons.forEach((svg, i) => {
        const d = document.createElement('div');
        d.className = 'item-select';
        d.innerHTML = `<svg viewBox="0 0 24 24">${svg}</svg>`;
        d.onclick = () => {
            if(selShapes.includes(i)) selShapes = selShapes.filter(x => x !== i);
            else selShapes.push(i);
            d.classList.toggle('active');
        };
        gGrid.appendChild(d);
    });

    document.getElementById('c').oninput = function() {
        document.getElementById('hexVal').innerText = this.value.toUpperCase();
        document.getElementById('hexVal').style.color = this.value;
        this.style.borderColor = this.value;
    };

    async function run(m) {
        const file = document.getElementById('f').files[0];
        const ans = document.getElementById('a').value;
        const num = document.getElementById('n').value;
        const hex = document.getElementById('c').value;

        if (!file || !ans || num.length < 12 || selSpecs.length === 0 || selShapes.length === 0) {
            alert("⚠️ CẢNH BÁO: Vui lòng nhập đủ 12 số và chọn ít nhất 1 ký tự + 1 hình!"); return;
        }

        const raw = ans + num + selSpecs.sort().join('') + selShapes.sort().join(',') + hex;
        const salt = CryptoJS.SHA512(hex + num + selSpecs.length).toString();

        const r = new FileReader();
        r.onload = function(e) {
            try {
                const k = CryptoJS.PBKDF2(raw, salt, { keySize: 512/32, iterations: 100000, hasher: CryptoJS.algo.SHA512 }).toString();
                const content = e.target.result;

                if (m === 'enc') {
                    const encrypted = CryptoJS.AES.encrypt(content, k).toString();
                    const blob = new Blob(["KSS_V3_BRIGHT|" + encrypted], {type: 'text/plain'});
                    const a = document.createElement('a');
                    a.href = URL.createObjectURL(blob);
                    a.download = `KSS_SECURE_${Date.now().toString().slice(-4)}.txt`;
                    a.click();
                    alert("✅ THÀNH CÔNG: File đã được tải xuống. Form sẽ tự động xóa sạch để bảo mật.");
                    location.reload(); 
                } else {
                    let data = content;
                    if(content.startsWith("KSS_V3_BRIGHT|")) data = content.split("|")[1];
                    const dec = CryptoJS.AES.decrypt(data, k);
                    const orig = dec.toString(CryptoJS.enc.Utf8);
                    if (!orig) throw new Error();
                    document.getElementById('o').value = orig;
                }
            } catch (err) { alert("❌ LỖI: Sai thông tin bảo mật hoặc file không hợp lệ!"); }
        };
        r.readAsText(file);
    }
</script>
