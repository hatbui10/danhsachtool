<script>
    const inputSlot = document.getElementById('input-slot');
    const logContainer = document.getElementById('log-container');
    const dataDisplay = document.getElementById('data-display');
    const fileUpload = document.getElementById('file-upload');
    
    let slots = ["", "", "", "", ""];
    let masterData = localStorage.getItem('masterData') || "";

    window.onload = () => {
        inputSlot.focus();
        if (masterData) renderDataCells();
    };

    document.addEventListener('click', () => inputSlot.focus());

    // Nhập ô 6 -> Dịch trái
    inputSlot.addEventListener('input', (e) => {
        const val = e.target.value.slice(-1);
        if (val >= '0' && val <= '9') {
            slots.shift();
            slots.push(val);
            
            for (let i = 0; i < 5; i++) {
                document.getElementById(`s${i}`).innerText = slots[i];
            }
            
            checkPattern();
        }
        inputSlot.value = "";
    });

    // Upload cộng dồn
    fileUpload.addEventListener('change', async (e) => {
        let newContent = "";
        for (let file of e.target.files) {
            newContent += await file.text();
        }
        masterData += newContent.replace(/\s/g, ""); // Loại bỏ khoảng trắng để xử lý chuỗi thuần số
        localStorage.setItem('masterData', masterData);
        renderDataCells();
    });

    // Vẽ mỗi số vào 1 ô
    function renderDataCells() {
        dataDisplay.innerHTML = "";
        const chars = masterData.split("");
        chars.forEach((char, index) => {
            const span = document.createElement('span');
            span.className = 'data-cell';
            span.id = `cell-${index}`;
            span.innerText = char;
            dataDisplay.appendChild(span);
        });
    }

    // Tìm kiếm
    function checkPattern() {
        const searchStr = slots.join("");
        if (searchStr.length < 5) return;

        // Tìm tất cả vị trí xuất hiện
        let index = masterData.indexOf(searchStr);
        if (index !== -1) {
            const timestamp = new Date().toLocaleTimeString();
            const logEntry = document.createElement('div');
            logEntry.className = 'log-item';
            logEntry.innerHTML = `<span style="color:var(--neon-color)">[${timestamp}]</span> Khớp dãy: <b style="color:#fff">${searchStr}</b>`;
            
            // Khi bấm log -> Cuộn tới ô đầu tiên của dãy đó
            logEntry.onclick = () => scrollToPattern(index, searchStr.length);
            logContainer.prepend(logEntry);
        }
    }

    function scrollToPattern(startIndex, length) {
        // Xóa highlight cũ
        document.querySelectorAll('.data-cell').forEach(el => el.classList.remove('highlight'));

        // Highlight các ô mới
        for (let i = 0; i < length; i++) {
            const cell = document.getElementById(`cell-${startIndex + i}`);
            if (cell) cell.classList.add('highlight');
        }

        // Cuộn tới
        const firstCell = document.getElementById(`cell-${startIndex}`);
        if (firstCell) {
            firstCell.scrollIntoView({ behavior: 'smooth', block: 'center' });
        }
    }

    function clearData() {
        if (confirm("Xóa toàn bộ dữ liệu?")) {
            masterData = "";
            localStorage.removeItem('masterData');
            dataDisplay.innerHTML = "";
            logContainer.innerHTML = "";
            slots = ["", "", "", "", ""];
            for (let i = 0; i < 5; i++) document.getElementById(`s${i}`).innerText = "";
        }
    }
</script>
